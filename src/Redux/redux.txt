previous versions of redux code -:
//store
const store = createStore(reducer);

//reducer
function reducer(state={initialvalue : 0} , action){
  return state;
}

// global state 

store.getState();


//action

store.dispatch(action);

action includes two arguments type and payload 

type - what type of action to perform like increment or decrement , addition or subtraction .
payload - payload is like the meta data or information about the action . 
for example -: increment valule by 10

{type: "increment", payload: "10"}
